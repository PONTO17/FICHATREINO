// Mude aqui se o nome da sua aba for diferente /** * Função para SALVAR dados na planilha (via requisição POST). */ function doPost(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const workoutData = JSON.parse(e.postData.contents); sheet.clear(); sheet.appendRow(["Timestamp", "WorkoutDataJSON"]); // Adiciona o cabeçalho const newRow = [new Date(), JSON.stringify(workoutData)]; sheet.appendRow(newRow); return ContentService .createTextOutput(JSON.stringify({ "status": "success", "message": "Data saved." })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } /** * Função para CARREGAR dados da planilha (via requisição GET). */ function doGet(request) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const lastRow = sheet.getLastRow(); // Se a planilha tiver mais do que apenas a linha do cabeçalho if (lastRow > 1) { const data = sheet.getRange(lastRow, 2).getValue(); // Pega o JSON da última linha return ContentService .createTextOutput(data) .setMimeType(ContentService.MimeType.JSON); } else { // Se a planilha estiver vazia, retorna um objeto vazio para o app criar os dados padrão return ContentService .createTextOutput(JSON.stringify({})) .setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService .createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
